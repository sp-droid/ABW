abwconfig.2 = { #STATS INFO
    type = country_event
    title = "abwconfig.2.t"
    desc = "abwconfig.2.desc"
    picture = scribe_tools
	
	option = { #ACTIVATE STATS
		name = abwconfig.2.a
		trigger = {
			has_variable = configflag_nostats
		}
		hidden_effect =  {
			remove_variable = configflag_nostats
			trigger_event = {
				id = "abwconfig.2"
			}
		}
	}
	
	option = { #DEACTIVATE STATS
		name = abwconfig.2.b
		trigger = {
			NOT = { has_variable = configflag_nostats }
		}
		hidden_effect =  {
			set_variable = configflag_nostats
			trigger_event = {
				id = "abwconfig.2"
			}
		}
	}
	
	option = { #GO BACK
		name = abwconfig.2.c
		hidden_effect = {
			trigger_event = {
				id = "abwconfig.1"
			}
		}
	}
}
